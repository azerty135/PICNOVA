    <script>
        const montants = [50, 100, 150, 200, 300, 500, 800, 1000, 1500, 2000, 3000, 5000, 8000, 10000];
        
        function renderGrid() {
            const grid = document.getElementById('grid-invest');
            montants.forEach(m => {
                grid.innerHTML += `<button onclick="invest(${m})">${m}$</button>`;
            });
        }

        function invest(amount) {
            const num = localStorage.getItem('numero_cico_actif') || "Non configuré";
            const conf = confirm("Investissement : " + amount + "$\n\nTransférez vers : " + num + "\n\nCliquez sur OK pour copier le numéro.");
            if (conf) {
                navigator.clipboard.writeText(num);
                alert("Numéro copié ! Allez sur Airtel Money.");
            }
        }

        function definirNumeroCICO() {
            const num = document.getElementById("nouveau-numero").value;
            if(num.length >= 10) {
                localStorage.setItem('numero_cico_actif', num);
                alert("Numéro mis à jour : " + num);
            } else {
                alert("Entrez un numéro valide.");
            }
        }

        function showPage(id) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.getElementById(id).classList.add('active');
        }

        renderGrid();
    </script>
